#include <stdio.h>
#include <iomanip.h>
#include <iostream.h>
#include <algorithm>
#include <queue>

using namespace std;

#define DMAX 101
#define TMAX 101

typedef struct Node{

	int Time;

	int Value;

}Node;

bool cmp(Node a,Node b)
{
	return a.Time<b.Time;
}

Node Dtable[DMAX];

int dp[DMAX][TMAX];

void main ()
{
	int T ,N;

	int i ,j ,temp;

	while(scanf("%d",&T)!=EOF)
	{
		scanf("%d",&N);

		for(i=1;i<=N;i++)
		{
			cin>>Dtable[i].Time>>Dtable[i].Value;
		}
		
		sort(Dtable+1 ,Dtable+1+N ,cmp);

		for(i=0;i<=N;i++)
		{
			for(j=0;j<=T;j++)
			{
				dp[i][j]=0;
			}
		}

		for(i=1;i<=N;i++)
		{
			for(j=1;j<=T;j++)
			{
				if(j>=Dtable[i].Time)
				{
					temp=Dtable[i].Value+dp[i-1][j-Dtable[i].Time];

					if(temp>dp[i][j-1])
					{
						dp[i][j]=temp;
					}
					else
					{
						dp[i][j]=dp[i][j-1];
					}
				}
				else
				{
					dp[i][j]=dp[i-1][j];
				}
			}
		}
		
 		cout<<dp[N][T]<<endl;
	}
}
