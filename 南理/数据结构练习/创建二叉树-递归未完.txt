#include <stdlib.h>
#include <stdio.h>

#define Elemtype char

typedef struct BiTNode{

	Elemtype data;

	struct BiTNode * lchild;

	struct BiTNode * rchild;

}BiTNode;

void CreateBiTree(BiTNode *t)
{
	Elemtype n;

	BiTNode lc ,rc;

	n=getchar();

	if(n!='#')
	{
		t->data=n;

		t->lchild=&lc;

		t->rchild=&rc;

		CreateBiTree(t->lchild);

		CreateBiTree(t->rchild);
	}
	else
	{
		t=NULL;
	}
}
