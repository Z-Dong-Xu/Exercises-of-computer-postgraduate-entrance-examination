#include <stdio.h>

#define MAX 20

typedef struct Stack{

	char *top;

	char *bottom;

	int size;

}SStack;

void InitStack(SStack &S)
{
	S.bottom=(char *)malloc(MAX*sizeof(char));

	S.top=S.bottom;

	S.size=MAX;
}

char Gettop(SStack &S)
{
	char n;

	if(S.top==S.bottom)
	{
		cout<<""<<endl;
	}

	n=*(S.top-1);

	return n;
}

void Push(SStack &S,char n)
{
	*(S.top)=n;

	S.top++;
}

char Pop(SStack &S )
{
	char n;

	S.top--;

	n=*(S.top);

	return n;
}

	while(c!='#')

		if(c=='('||c=='['||c=='{')
		{
			Push(S,c);

		if(c==')')
		{
			if(Gettop(S)=='(')
			{

		if(c=='}')
