#include <iostream.h>
#include <math.h>

int S( int n);

void main ()
{
	int num ,flag;

	cout<<"输入"<<endl;

	cin>>num;

	flag=S(num);

	if(flag)
		cout<<"不是"<<endl;
	else
		cout<<"是"<<endl;
}

int S( int n)
{
	int i,f=0;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{	f=1;break;}
		else
			f=0;
	}

return f;
}
