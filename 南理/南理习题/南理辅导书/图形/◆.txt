#include <iostream.h>

#include <iomanip.h>

void main()
{
	int n , m;

	int i , j;

	cout<<"ÊäÈë"<<endl;

	cin>>n;

	for(i=0;i<n;i++)
	{
		if(i<=n/2)
		{
			m=i*2+1;
		}
		else
			m=1+(n-i-1)*2;

		cout<<setw(1*(n-m+1));

		for(j=0;j<m;j++)
		{
			cout<<"*"<<setw(2);
		}
		cout<<endl;
	}
	cout<<endl;
}
