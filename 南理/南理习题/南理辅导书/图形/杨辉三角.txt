void fun(int a[][100],int n)
{
	int i,j;
	for(i=0;i<n;i++)
	{
		a[i][i]=1;
		a[i][0]=1;
	}
	for(i=1;i<n;i++)
	{
		for(j=1;j<i;j++)
		{
			a[i][j]=a[i-1][j-1]+a[i-1][j];
		}
	}
}
