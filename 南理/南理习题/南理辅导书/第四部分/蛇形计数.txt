#include <iostream.h>
#include <iomanip.h>

void main ()
{
	int num ,i ,j ,a[100][100];

	cout<<"ÊäÈë"<<endl;

	cin>>num;

	for(i=1;i<=num;i++)
	{
		if(i%2==0)
		{
			a[i][1]=(i*(i-1))/2+1;
			for(j=1;j<i;j++)
			{
				a[i-j][j+1]=a[i-j+1][j]+1;
			}
		}
		else
		{
			a[1][i]=(i*(i-1))/2+1;
			for(j=1;j<i;j++)
			{
				a[j+1][i-j]=a[j][i-j+1]+1;
			}
		}
	}

	for(i=1;i<=num;i++)
	{
		for(j=1;j<=num-i+1;j++)
		{
			cout<<setw(8)<<a[i][j];
		}
		cout<<endl;
	}
}
